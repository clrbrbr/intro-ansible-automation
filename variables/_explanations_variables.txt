- show_ip_routes.yaml

---

- hosts: all
  connection: network_cli
  gather_facts: no

  tasks:
      - name: Checking connectivity
        ignore_errors: True
        ios_command:
          commands:
            - show ip route
        register: ios_output

      - name: Show output			Affiche la sortie de la commande sous forme de liste de lignes avec "stdout_lines"
        debug: 
          msg: "{{ ios_output.stdout_lines }}"
