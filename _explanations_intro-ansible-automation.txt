- ansible.cfg				Définit la configuration d'Ansible: comportement par défaut SSH, timeout, sécurité, logs, collecte des facts, etc.

[defaults]				
gathering = explicit			Désactive la collecte automatique des facts au début des playbooks, Les facts ne seront récupérés seulement si "gather_facts: true" ou les module "setup" est utilisé
host_key_checking = True		Active la vérification des clés SSH des hôtes distants, Ansible refus de se connecter à un serveur inconnu ou dont la clé SSH a changé
timeout = 10				Définit le délai maximum de connexion SSH à 10 secondes avant échec
deprecation_warnings = False		Masque les avertissements concernant les fonctionnalités Ansible obsolètes (deprecated)
retry_files_enabled = False		Désactive la création des fichiers ".retry" contenant la liste des hôtes en échec après l'exécution d'un playbook
