- sh_ip_routes_conditionals.yaml	Playbook Ansible qui vérifie l'exécution de la commande "show ip route" sur tous les équipements et affiche un résultat conditionnel

---

- hosts: all						Exécute le playbook sur tous les équipements de l'inventaire
  connection: network_cli				Utilise une connexion CLI réseau Cisco via SSH
  gather_facts: no					Désactive la collecte automatique d'informations

  tasks:
      - name: Checking connectivity			Tâche principale de vérification
        ignore_errors: True				Continue l'exécution même en cas d'erreur
        ios_command:
          commands:
            - show ip route				Exécute la commande Cisco "show ip route"
        register: ios_output				Stocke le résultat dans la variable "ios_output"

      #   debug:

      # - name: Verification of output (success)	Message en cas de succès
      #   debug: 
      #     msg: "{{ 'Success' }}"
      #   when: ios_output.failed == False


      - name: Verification of output			Affiche une message unique "Success" si la commande a réussi, ou "Failed with message:..." si la commande a échoué
        debug: 
          msg: "{{ 'Failed with message: ' ~ ios_output.msg if ios_output.failed else 'Success' }}"		Version simplifiée utilisant une condition "if/else" dans Jinja2
															{{ ... }}					Evalue une expression dans Ansible
															ios.output.failed				Vrai si la commande a échoué, Faux si elle a réussi
															'Failed with message: ' ~ ios_output.msg	"~" permet la concaténation de chaînes en Jinja2
